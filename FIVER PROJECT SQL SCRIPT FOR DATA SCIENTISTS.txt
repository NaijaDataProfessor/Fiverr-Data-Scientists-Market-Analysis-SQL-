--- Words Count in Gig Title Column
WITH RECURSIVE words AS (
    SELECT
        LOWER(
            TRIM(
                SUBSTRING_INDEX(gigtitle, ' ', 1)
            )
        ) AS word,
        SUBSTRING(gigtitle, LENGTH(SUBSTRING_INDEX(gigtitle, ' ', 1)) + 2) AS rest
    FROM dataset
    WHERE gigtitle IS NOT NULL

    UNION ALL

    SELECT
        LOWER(
            TRIM(
                SUBSTRING_INDEX(rest, ' ', 1)
            )
        ),
        SUBSTRING(rest, LENGTH(SUBSTRING_INDEX(rest, ' ', 1)) + 2)
    FROM words
    WHERE rest <> ''
)

SELECT
    word,
    COUNT(*) AS word_count
FROM words
WHERE word <> ''
GROUP BY word
ORDER BY word_count DESC;

--- Title with the higest order completed on fiver
SELECT gigtitle AS gig_title, SUM(ordercompleted) AS order_completed, price, rating, badge
FROM dataset
GROUP BY gig_title, price, rating, badge
ORDER BY order_completed DESC
LIMIT 20
